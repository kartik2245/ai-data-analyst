DATABASE SCHEMA

================================================================================



TABLE: categories
--------------------------------------------------------------------------------
category_id (uuid) [PRIMARY KEY]
category_name (character varying)
description (text)

TABLE: customers
--------------------------------------------------------------------------------
customer_id (uuid) [PRIMARY KEY]
customer_code (character varying)
customer_name (character varying)
email (character varying)
phone (character varying)
city (character varying)
state (character varying)
registration_date (date)
is_active (boolean)

TABLE: inventory
--------------------------------------------------------------------------------
inventory_id (uuid) [PRIMARY KEY]
product_id (uuid) -> products.product_id
quantity_available (integer)
warehouse_location (character varying)
last_restock_date (date)

TABLE: order_items
--------------------------------------------------------------------------------
order_item_id (uuid) [PRIMARY KEY]
order_item_code (character varying)
order_id (uuid) -> orders.order_id
product_id (uuid) -> products.product_id
quantity (integer)
unit_price (numeric)
discount_percentage (numeric)
total_price (numeric)

TABLE: orders
--------------------------------------------------------------------------------
order_id (uuid) [PRIMARY KEY]
order_code (character varying)
customer_id (uuid) -> customers.customer_id
order_date (date)
order_status (character varying)
total_amount (numeric)
shipping_address (text)
shipping_city (character varying)
shipping_state (character varying)
shipping_pincode (integer)
shipping_country (character varying)

TABLE: payments
--------------------------------------------------------------------------------
payment_id (uuid) [PRIMARY KEY]
payment_code (character varying)
order_id (uuid) -> orders.order_id
payment_method (character varying)
payment_status (character varying)
transaction_id (character varying)
payment_date (date)
amount_paid (numeric)

TABLE: products
--------------------------------------------------------------------------------
product_id (uuid) [PRIMARY KEY]
category_id (uuid) -> categories.category_id
product_name (character varying)
brand (character varying)
description (text)
price (numeric)
sku (character varying)
rating (numeric)
is_active (boolean)

TABLE: reviews
--------------------------------------------------------------------------------
review_id (uuid) [PRIMARY KEY]
customer_id (uuid) -> customers.customer_id
product_id (uuid) -> products.product_id
rating (integer)
review_title (character varying)
review_text (text)
review_date (date)
is_verified_purchase (boolean)

TABLE: shipments
--------------------------------------------------------------------------------
shipment_id (uuid) [PRIMARY KEY]
shipment_code (character varying)
order_id (uuid) -> orders.order_id
courier_name (character varying)
tracking_number (character varying)
shipment_status (character varying)
shipped_date (date)
estimated_delivery_date (date)
delivered_date (date)
